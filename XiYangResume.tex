\resheading{Selected Open Source Projects}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{itemize}
\item
  \ressubheadingTwo{SHIM}{\texttt{\color{blue} github.com/shimProfiler/SHIM}}
  \begin{itemize}
    \resitem{SHIM is a open source high frequency continuous profiler. I designed,
      implemented, and maintain the project.}
  \end{itemize}
\item
  \ressubheadingTwo{Jikes RVM on
    OpenJDK}{\texttt{\href{http://hg.code.sourceforge.net/p/jikesrvm/code/shortlog/1c39f36bc842}{\color{blue}
        RVM-549-OpenJDK}}}    
  \begin{itemize}
    \resitem{Jikes RVM is not able to run many new and important applications
      because it is using the GNU Classpath library and the library is
      not in active development anymore. This project attempts to port the OpenJDK
      library to Jikes RVM to fix the problem.}
  \end{itemize}
\item
  \ressubheadingTwo{RTEMS on the Cortex-M3 CPU of OMAP4 SoC}{\texttt{\color{blue} https://github.com/yangxi/omap4m3}}
  \begin{itemize}
    \resitem{The OMAP4 SoC has the potential to
      become a nice research platform for researchers who are interested in the
      heterogeneous system, since it has two big Cortex-A9
      cores and two small Cortex-M3 cores. However, because there is no open
      source RTOS running on the small cores, it is hard to try ideas on the
      platform. This project ported the RTEMS to the
      small M3 cores. The RTEMS is loaded by and communicates with the Linux
      kernel running on the big A9 cores.}
    \end{itemize}
\end{itemize}
